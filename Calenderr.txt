bool CheckADay(int Day, int Month, int Year)
{
	if (Year > 1900)
	{
		if (Month > 0 && Month <= 12)
		{
			switch (Month)
			{
			case 1:
			case 3:
			case 5:
			case 7:
			case 8:
			case 10:
			case 12:
				if (Day > 0 && Day <= 31)
				{
					return true;
				}
				else
				{
					return false;
				}
			case 2:
				if (Year % 400 == 0 || Year % 4 == 0 && Year % 100 != 0)
				{
					if (Day > 0 || Day <= 29)
					{
						return true;
					}
				}
				else
				{
					if (Day > 0 && Day <= 28)
					{
						return true;
					}
					else
					{
						return false;
					}
				}
			default:
				if (Day > 0 && Day <= 30)
				{
					return true;
				}
				else
				{
					return false;
				}
			}
		}
		else
		{
			return false;
		}
	}
	else
	{
		return false;
	}
}
int dayofweek(int day, int month, int year)
{
	int a = (14 - month) / 12;
	int y = year - a;
	int m = month + 12 * a - 2;
	int dayofweeky;
	dayofweeky = (day + y + y / 4 - y / 100 + y / 400 + (31 * m) / 12) % 7;
	return dayofweeky;
}
